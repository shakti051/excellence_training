Note:- Stream is like a pipe in which data added from sink n output collected with stream controller listen.

1. program :-add simply numbers in sink
import 'dart:async';
void main() {
  final controller = StreamController();
  //data added using sink
    controller.sink.add(1);
    controller.sink.add(2);
    controller.sink.add(3);
  // data out is checked by listen
  
  controller.stream.listen((val){
    print(val);
  });
}

2. program :- add if number is less than 5

import 'dart:async';

  void addLessThanFive(StreamController controller, int value) {
  if (value < 5) {
    controller.sink.add(value);
  } else {
    controller.sink.addError(StateError('$value is not less than five'));
  }
}

  void main() {
  final controller = StreamController();
  //data added using sink
  addLessThanFive(controller, 1);
  addLessThanFive(controller, 2);
  addLessThanFive(controller, 3);
  addLessThanFive(controller, 4);
  addLessThanFive(controller, 5);
   controller.close();

  // data out is checked by listen
  controller.stream.listen((val) {
    print(val);
  }, onError: (error) {
    print(error);
  }, onDone:() {
    print('done');
  });
}
